264         /**
265          * 为目标元素添加事件监听器
266          * @name baidu.event.on
267          * @function
268          * @grammar baidu.event.on(element, type, listener)
269          * @param {HTMLElement|string|window} element 目标元素或目标元素id
270          * @param {string} type 事件类型
271          * @param {Function} listener 需要添加的监听器
272          * @remark
273          *
274         1. 不支持跨浏览器的鼠标滚轮事件监听器添加<br>
275         2. 改方法不为监听器灌入事件对象，以防止跨iframe事件挂载的事件对象获取失败
276 
277          * @shortcut on
278          * @meta standard
279          * @see baidu.event.un
280          *
281          * @returns {HTMLElement|window} 目标元素
282          */
283         baidu.event.on = function (element, type, listener) {
284             type = type.replace(/^on/i, '');
285             element = baidu.dom._g(element);
286 
287             var realListener = function (ev) {
288                     // 1. 这里不支持EventArgument,  原因是跨frame的事件挂载
289                     // 2. element是为了修正this
290                     listener.call(element, ev);
291                 },
292                 lis = baidu.event._listeners,
293                 filter = baidu.event._eventFilter,
294                 afterFilter, realType = type;
295             type = type.toLowerCase();
296             // filter过滤
297             if (filter && filter[type]) {
298                 afterFilter = filter[type](element, type, realListener);
299                 realType = afterFilter.type;
300                 realListener = afterFilter.listener;
301             }
302 
303             // 事件监听器挂载
304             if (element.addEventListener) {
305                 element.addEventListener(realType, realListener, false);
306             } else if (element.attachEvent) {
307                 element.attachEvent('on' + realType, realListener);
308             }
309 
310             // 将监听器存储到数组中
311             lis[lis.length] = [element, type, listener, realListener, realType];
312             return element;
313         };
314

316         baidu.on = baidu.event.on;

318         /**
319          * 为目标元素移除事件监听器
320          * @name baidu.event.un
321          * @function
322          * @grammar baidu.event.un(element, type, listener)
323          * @param {HTMLElement|string|window} element 目标元素或目标元素id
324          * @param {string} type 事件类型
325          * @param {Function} listener 需要移除的监听器
326          * @shortcut un
327          * @meta standard
328          * @see baidu.event.on
329          *
330          * @returns {HTMLElement|window} 目标元素
331          */
332         baidu.event.un = function (element, type, listener) {
333             element = baidu.dom._g(element);
334             type = type.replace(/^on/i, '').toLowerCase();
335 
336             var lis = baidu.event._listeners,
337                 len = lis.length,
338                 isRemoveAll = !listener,
339                 item, realType, realListener;
340 
341             //如果将listener的结构改成json
342             //可以节省掉这个循环，优化性能
343             //但是由于un的使用频率并不高，同时在listener不多的时候
344             //遍历数组的性能消耗不会对代码产生影响
345             //暂不考虑此优化
346             while (len--) {
347                 item = lis[len];
348 
349                 // listener存在时，移除element的所有以listener监听的type类型事件
350                 // listener不存在时，移除element的所有type类型事件
351                 if (item[1] === type && item[0] === element && (isRemoveAll || item[2] === listener)) {
352                     realType = item[4];
353                     realListener = item[3];
354                     if (element.removeEventListener) {
355                         element.removeEventListener(realType, realListener, false);
356                     } else if (element.detachEvent) {
357                         element.detachEvent('on' + realType, realListener);
358                     }
359                     lis.splice(len, 1);
360                 }
361             }
362 
363             return element;
364         };

887     function attr(elem, name, value) {
888         if (!name || name.constructor != String) return '';
889         name = {
890             'for': 'htmlFor',
891             'class': 'className'
892         }[name] || name;
893         if (typeof value != 'undefined') {
894             elem[name] = value;
895             if (elem.setAttribute) {
896                 elem.setAttribute(name, value);
897             }
898         }
899         return elem[name] || elem.getAttribute(name) || '';

908     function create(tag, attr) {
909         var e = document.createElement(tag);
910         attr = attr || {};
911         // 设置属性
912         for (var name in attr) {
913             value = attr[name];
914             name = result()[name] || name;
915             if (name == "style") {
916                 e.style.cssText = value;
917                 continue;
918             }
919             if (e.setAttribute) {
920                 e.setAttribute(name, value);
921             } else {
922                 try {
923                     e[name] = value;
924                 } catch (e) {}
925             }
926         }
927         return e;

955         var pos = {

957             top: 0

966     /**
967      * 在目标元素的指定位置插入HTML代码，
968      * 闭包，对外不暴露
969      * @ignore
970      * @param {HTMLElement|string} element 目标元素或目标元素的id
971      * @param {String} position 插入html的位置信息
972      *     取值为beforeBegin、afterBegin、beforeEnd或afterEnd，大小写不敏感
973      * @param {String} html 要插入的html
974      * @return {HTMLElement} 目标元素
975      */
976 
977     function insertHTML(element, position, html) {
978         var range, begin;
979         if (element.insertAdjacentHTML) {
980             element.insertAdjacentHTML(position, html);
981         } else {
982             range = element.ownerDocument.createRange();
983             // FF下range的位置设置错误可能导致创建出来的fragment在插入dom树之后html结构乱掉
984             // 改用range.insertNode来插入html
985             position = position.toUpperCase();
986             if (position == 'AFTERBEGIN' || position == 'BEFOREEND') {
987                 range.selectNodeContents(element);
988                 range.collapse(position == 'AFTERBEGIN');
989             } else {
990                 begin = position == 'BEFOREBEGIN';
991                 range[begin ? 'setStartBefore' : 'setEndAfter'](element);
992                 range.collapse(begin);
993             }
994             range.insertNode(range.createContextualFragment(html));
995         }
996         return element;
997     }
998     /**

1000      * @param {String}  url
1001      * @param {Function} callback

1003      */
1004    function scriptRequest(url,callback){
1005         var script = document.createElement('script');
1006         script.setAttribute('src', url);
1007         script.setAttribute('type', 'text/javascript');
1008         script.setAttribute('charset', 'gbk');
1009         // 脚本加载完成后进行移除
1010         if (script.addEventListener) {
1011           script.addEventListener('load', function(e) {
1012             var t = e.target;
1013             t.parentNode.removeChild(t);
1014             if(callback) callback();
1015           }, false);
1016         }
1017         else if (script.attachEvent) {
1018           script.attachEvent('onreadystatechange', function(e) {
1019             var t = window.event.srcElement;
1020             if (t && (t.readyState == 'loaded' || t.readyState == 'complete')) {
1021               t.parentNode.removeChild(t);
1022               if(callback) callback();
1023             }
1024           });
1025         }
1026         // 使用setTimeout解决ie6无法发送问题
1027         setTimeout(function() {
1028           document.getElementsByTagName('head')[0].appendChild(script);
1029           script = null;
1030         }, 1);
1031     }
